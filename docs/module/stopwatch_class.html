Contains code timing capabilities

Example usage

program stopwatch_test
use variableKind, only: i32
use Stopwatch_Class, only: Stopwatch
type(Stopwatch) :: clk
integer(i32) :: i, N
call clk%start('Some Title')
N = 100000
do i = 1, N
  Compute some stuff
enddo
call clk%stop()
call clk%elapsed()
write(*,'(a)') 'Finished on '//clk%datetime()
end program
