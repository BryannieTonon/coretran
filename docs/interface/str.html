public interface str

Interface to print a string representation of a number The output format options can be set using the printOptions class

Example usage

use variableKind
use m_strings
real(r64) :: arr(20)
integer(i32) :: i
integer(i32) :: j(5)
arr = [(dble(i), i = 1,20)]
i = 12
j = [0, 1, 2, 3, 4]
write(*,'(a)') str(i)//str(j)
write(*,'(a)') str(j)//str(arr)
write(*,'(a)') str(arr)
printOptions%threshold = 0
write(*,'(a)') str(arr)
